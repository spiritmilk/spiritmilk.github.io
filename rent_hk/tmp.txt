function filterData(data, columnFilters) {
  return data.filter(row => {
    for (const column in columnFilters) {
      const filterValue = columnFilters[column];
      if (filterValue && row[column] !== filterValue) {
        return false;
      }
    }
    return true;
  });
}

// 示例数据
const data = [
  { name: "John", age: 25, city: "New York" },
  { name: "Alice", age: 30, city: "London" },
  { name: "Bob", age: 25, city: "Paris" },
  { name: "Jane", age: 35, city: "New York" }
];

// 用户选择的筛选条件
const columnFilters = {
  age: 25,
  city: "New York"
};

// 对数据进行筛选
const filteredData = filterData(data, columnFilters);

// 显示筛选结果
console.log(filteredData);
